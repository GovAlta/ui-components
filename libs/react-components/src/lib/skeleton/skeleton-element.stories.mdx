# GoASkeletonElement

A skeleton load element provides visual feedback to users while a content heavy page or element is loading.

## Variations

export const ImageContent = (args) => (
  <div>
    <span>
      <b>Content with image</b>
    </span>
    <GoASkeletonImageContent {...args} />
  </div>
);

export const Content = (args) => (
  <div>
    <span>
      <b>Content with a title</b>
    </span>
    <GoASkeletonContent {...args} />
  </div>
);

export const Card = (args) => (
  <div>
    <span>
      <b>Card content</b>
    </span>
    <div style={{ display: 'flex', gap: '8px' }}>
      <GoALoadingCardWithImage {...args} />
      <GoALoadingCardWithImage {...args} />
      <GoALoadingCardWithImage {...args} />
    </div>
  </div>
);

export const ColumnContent = (args) => (
  <div>
    <span>
      <b>Content without a title</b>
    </span>
    <GoASkeletonGridColumnContent {...args} />
  </div>
);
