# GoAElementLoader

A element loading indicator which will not prevent user input.

## Overview

export const Template = (args) => (
  <div>
    <GoAElementLoader {...args} />
  </div>
);

export const TemplateElement = (args) => (
  <div>
    <GoAButton buttonType="primary" buttonSize="normal">
      Loading please wait...
      <GoAElementLoader {...args} />

  </div>
);
