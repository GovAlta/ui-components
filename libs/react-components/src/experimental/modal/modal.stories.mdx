# GoAModal

A dialog that appears on top of the main content and moves the system into a special mode requiring user interaction. This dialog disables the main content until the user explicitly interacts with the modal dialog.

## Overview

export const Template = (args) => {
  const [isOpen, setOpen] = useState(false);
  const [clickCount, setClickCount] = useState(0);
  return (
    <>
      <button onClick={() => setOpen(true)}>Open Modal</button>
      <GoAModal testId="modal" isOpen={isOpen}>
        <GoAModalTitle testId="title">{args.title}</GoAModalTitle>
        <GoAModalContent testId="content">{args.content}</GoAModalContent>
        <GoAModalActions>
          <GoAButton
            buttonType="tertiary"
            type="button"
            onClick={() => setOpen(false)}
          >
            Cancel
          </GoAButton>
          <GoAButton
            buttonType="primary"
            type="submit"
            onClick={() => {
              setClickCount(clickCount + 1);
              action('clicked save')(clickCount);
            }}
          >
            Save
          </GoAButton>
        </GoAModalActions>
      </GoAModal>
    </>
  );
};

export const ModalFormTemplate = (args) => {
  const [formState, setFormState] = useState({});
  const [isOpen, setOpen] = useState(false);
  const onChange = (name, value) => {
    setFormState({
      ...formState,
      [name]: value,
    });
  };
  const data = [
    { label: 'option 1', value: 'option1' },
    { label: 'option 2', value: 'option2' },
    { label: 'option 3', value: 'option3' },
    { label: 'option 4', value: 'option4' },
  ];
  return (
    <>
      <button onClick={() => setOpen(true)}>Open Modal</button>
      <GoAModal testId="modal" isOpen={isOpen}>
        <GoAModalTitle testId="title">{args.title}</GoAModalTitle>
        <GoAModalContent testId="content">
          <GoAForm>
            <GoAFormItem>
              <label>First Name</label>
              <GoAInput
                type="text"
                name="firstName"
                value={formState.firstName}
                onChange={onChange}
              />
            </GoAFormItem>
            <GoAFormItem>
              <label>Middle Name</label>
              <GoAInput
                type="text"
                name="middleName"
                value={formState.middleName}
                onChange={onChange}
              />
            </GoAFormItem>
            <GoAFormItem>
              <label>Colors</label>
              <GoADropdown
                name="favColours"
                onChange={onChange}
                selectedValues={formState.favColours}
              >
                {data.map((item) => (
                  <GoADropdownOption label={item.label} value={item.value} />
                ))}
              </GoADropdown>
            </GoAFormItem>
            <GoAFormItem>
              <label>Last Name</label>
              <GoAInput
                type="text"
                name="lastName"
                value={formState.lastName}
                onChange={onChange}
              />
            </GoAFormItem>
            <GoAFormItem>
              <label>Favourite Music</label>
              <GoARadioGroup
                name="music"
                onChange={onChange}
                value={formState.music}
              >
                <GoARadio label="Blues" value="blues" />
                <GoARadio label="Rock" value="rock" />
                <GoARadio label="Country" value="country" />
              </GoARadioGroup>
            </GoAFormItem>
            <GoAFormItem helpText="We won't spam you...">
              <label>Email</label>
              <GoAInput
                type="text"
                name="email"
                value={formState.email}
                onChange={onChange}
              />
            </GoAFormItem>
            <GoAFormItem helpText="Enter lots of text here">
              <label>Label</label>
              <GoATextArea
                name="textarea"
                value={formState.textarea}
                onChange={onChange}
              />
            </GoAFormItem>
          </GoAForm>
        </GoAModalContent>
        <GoAModalActions>
          <GoAButton
            buttonType="tertiary"
            type="button"
            onClick={() => setOpen(false)}
          >
            Cancel
          </GoAButton>
          <GoAButton
            buttonType="primary"
            type="submit"
            onClick={() => {
              setOpen(false);
            }}
          >
            Save
          </GoAButton>
        </GoAModalActions>
      </GoAModal>
    </>
  );
};

### Actions pinned to the bottom of the modal

```jsx
function MyComponent() {
  const [isOpen, setOpen] = useState(false);
  const [clickCount, setClickCount] = useState(0);
  return (
    <div>
      <button onClick={() => setOpen(true)}>Open Modal</button>
      <GoAModal isOpen={isOpen}>
        <GoAModalTitle>Form</GoAModalTitle>
        <GoAModalContent>
          <GoAForm>
            <GoAFormItem>
              <label>Name</label>
              <input type="text" name="name" />
            </GoAFormItem>
            <GoAFormItem>
              <label>Description</label>
              <textarea name="description" />
            </GoAFormItem>
          </GoAForm>
        </GoAModalContent>
        <GoAModalActions>
          <GoAButton
            buttonType="tertiary"
            type="button"
            onClick={() => setIsOpen(false)}
          >
            Cancel
          </GoAButton>
          <GoAButton buttonType="primary" type="submit">
            Save
          </GoAButton>
        </GoAModalActions>
      </GoAModal>
    </div>
  );
}
```

## Usage guidelines

The modal is designed having three sections, a title, content and actions. The action section should contain any buttons or other controls
that require user interaction, however if necessary they can be placed in the content section and require the user to scroll the bottom
before they can be seen.
