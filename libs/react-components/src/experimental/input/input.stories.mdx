## Overview

export const Template = (args) => {
  const [state, setState] = useState({});
  const onChange = (name, value) => {
    setState({ ...state, [name]: value });
  };
  return (
    <>
      <State
        title="Input Types"
        description="The GoAInput component handles all html input types styled to the GoA UI guidelines"
        state={state}
      >
        <GoAInputText
          {...args}
          onChange={onChange}
          name="basic"
          leadingIcon="personCircle"
        />
        <GoAInputPassword {...args} onChange={onChange} name="password" />
        <GoAInputDate {...args} onChange={onChange} name="date" />
        <GoAInputTime {...args} onChange={onChange} name="time" />
        <GoAInputDateTime {...args} onChange={onChange} name="datetime" />
        <GoAInputEmail {...args} onChange={onChange} name="email" />
        <GoAInputSearch {...args} onChange={onChange} name="search" />
        <GoAInputUrl {...args} onChange={onChange} name="url" />
        <GoAInputTel {...args} onChange={onChange} name="tel" />
        <GoAInputFile {...args} onChange={onChange} name="file" />
        <GoAInputMonth {...args} onChange={onChange} name="month" />
        <GoAInputNumber {...args} onChange={onChange} name="number" />
        <GoAInputRange {...args} onChange={onChange} name="range" />
      </State>
    </>
  );
};
