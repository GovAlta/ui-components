# GoAInput

## Overview

export const Template = (args) => {
  const [state, setState] = useState({});
  const onChange = (name, value) => {
    setState({ ...state, [name]: value });
  };
  return (
    <>
      <State
        title="Input Types"
        description="The GoAInput component handles all html input types styled to the GoA UI guidelines"
        state={state}
      >
        <GoAInput {...args} onChange={onChange} name="basic" leadingIcon="personCircle" />
        <GoAInput {...args} onChange={onChange} type="password" name="password" />
        <GoAInput {...args} onChange={onChange} type="date" name="date" />
        <GoAInput {...args} onChange={onChange} type="time" name="time" />
        <GoAInput {...args} onChange={onChange} type="datetime-local" name="datetime" />
      </State>
    </>
  );
};
