# GoAElementLoader

A element loading indicator which will not prevent user input.

#### Basic

<Canvas>
  <Story
    name="Basic"
    height={200}
    args={{

      size: 'default',
      baseColour: '#c8eef9',
      spinnerColour: '#0070c4',
    }}
  >
    {(args, argTypes) => {
      return {
        render: function (createElement) {
          return createElement(GoAElementLoader, { props: { ...args } }, null);
        },

    }}

</Canvas>

## Examples

```
<template>
  <div id="app">
    <goa-button
      buttonType="'primary || 'secondary' || 'tertiary'"
      buttonSize="'small' || 'normal'"
      title="Hovering over"
      v-on:click="() => function()"
    >
    <goa-element-loader
      visible= true|false
      size= 'default'|'small'
      baseColour= '#0070c4'
      spinnerColour= '#c8eef9'
    />
    </goa-button>
  </div>
</template>

<script>
import { GoAButton } from '@abgov/vue-components';

export default {
  name: "App",
  components: {
    'goa-element-loader': GoAElementLoader
  }
};
</script>
```
