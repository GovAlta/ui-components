# Accordion

<GoACallout>Accordion containers enable multiple content sections to be displayed in a limited space and collapsed or expanded by the user. You can create hierarchy of information by hiding secondary content inside collapsed expand containers.
</GoACallout>

#### Accordion Group

<GoACallout>The open prop on accordion component can be used to control a group of accordions.</GoACallout>

export const AccordionGroupTemplate = () => {
  const [open, setOpen] = useState(false);
  const buttonState = open ? "Hide all" : "Show all";
  return (
    <>
      <GoAButton type="tertiary" onClick={() => setOpen(!open)}>
        {buttonState} sections
      </GoAButton>
      <br />
      <GoAAccordion heading="Heading" open={open}>
        <p>
        Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor
        incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis
        nostrud exercitation ullamco laboris nisi
        </p>
      </GoAAccordion>
      <GoAAccordion heading="Heading" open={open}>
        <p>
        Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor
        incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis
        nostrud exercitation ullamco laboris nisi
        </p>
      </GoAAccordion>
      <GoAAccordion heading="Heading" open={open}>
        <p>
        Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor
        incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis
        nostrud exercitation ullamco laboris nisi
        </p>
      </GoAAccordion>
    </>
  );
};
