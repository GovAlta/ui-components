# Dropdown

### Dropdowns hide a long list of options, arranged vertically. A single select menu list is revealed upon interaction with this component.

[Figma design component](https://www.figma.com/file/3pb2IK8s2QUqWieH79KdN7/%E2%9D%96-Components-%7C-DIO-DS?node-id=700%3A7316)

export const Template = (args) => {
  const [value, setValue] = useState(args.value ? args.value : "");
  function onChange(name, value) {
    console.log(name, value);
    setValue(value);
  }
  return (
    <GoADropdown
      {...args}
      filterable={args.filterable}
      value={value}
      onChange={onChange}
    >
      <GoADropdownItem value="red" label="Red" />
      <GoADropdownItem value="green" label="Green" />
      <GoADropdownItem value="blue" label="Blue" />
    </GoADropdown>
  );
};

export const TemplateAccessibility = (args) => {
  const [value, setValue] = useState(args.value);
  function onChange(name, value) {
    console.log(name, value);
    setValue(value);
  }
  return (
    <GoAFormItem label="Color" fieldId={args.id ? args.id : "color"}>
      <GoADropdown {...args} value={value} onChange={onChange}>
        <GoADropdownItem value="red" label="Red" />
        <GoADropdownItem value="green" label="Green" />
        <GoADropdownItem value="blue" label="Blue" />
      </GoADropdown>
    </GoAFormItem>
  );
};

<GoACallout type="important">
  Include an 'arialabel' attribute whenever you’re hiding a label or not using
  GoAFormItem. This 'arialabel' should provide a succinct, meaningful
  description of the dropdown’s purpose or the information it represents.
</GoACallout>

<GoACallout type="information">
  Include an 'arialabelledby' for the GoADropdown that matches with the 'id' for the GoAFormItem. It
  ensures that assistive technology can refer the correct label when presenting
  a form control of the dropdown.
</GoACallout>

            [value]="reactiveFormCtrl.value">

            *ngFor="let color of colors"
            [value]="color"
            [label]="color"
