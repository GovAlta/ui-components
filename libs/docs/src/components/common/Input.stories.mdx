### Basic

export const InputTemplate = () => {
  const [value, setValue] = useState('');
  return (
    <>
      <GoAInputText
        name="firstname"
        value={value}
        onChange={(name, value) => {
          setValue(value);
          console.log(name, value);
        }}
      />
    </>
  );
};
