# Three column layout

### A three column layout to use as a starting point for your page. Typically, this is used for a side navigation alongside the main content of a page and a side menu on the right.

[Figma design component](https://www.figma.com/file/ylmHeuDMfxnDBnP1VaQYz8/%E2%9D%96-Styles-and-Content-%7C-DDI?node-id=3787%3A103173)

---

#### Properties

  <Prop
    name="leftcolumnwidth"
    type="string"
    lang="angular"
    description="Set the width of the left column. Values can be: px, %, ch etc."

  />
  <Prop
    name="leftColumnWidth"
    type="string"
    lang="react"
    description="Set the width of the left column. Values can be: px, %, ch etc."
    defaultValue="256px"
  />
  <Prop
    name="rightcolumnwidth"
    type="string"
    lang="angular"
    description="Set the width of the right column."
    defaultValue="256px"
  />
  <Prop
    name="rightColumnWidth"
    type="string"
    lang="react"
    description="Set the width of the right column."
    defaultValue="256px"
  />
  <Prop
    name="maxcontentwidth"
    type="string"
    lang="angular"
    description="Set the maximum width of all three columns. Values can be: px, %, ch etc."
    defaultValue="100%"
  />
  <Prop
    name="maxContentWidth"
    type="string"
    lang="react"
    description="Set the maximum width of all three columns. Values can be: px, %, ch etc."
    defaultValue="100%"
  />

    name="header"
    lang="angular"
    type="slot"

  />
  <Prop
    name="footer"

  />
  <Prop
    name="footer"

    type="ReactNode"
    description="Space for GoAAppFooter component"
  />
  <Prop
    name="nav"
    lang="angular"

  />
  <Prop

  />
  <Prop

  />
  <Prop

#### Basic

<Story name="Basic">
  <GoACallout type="information">
    For situations where default values of 256px for the nav column and side menu,
    and 100% for main content.
  </GoACallout>
  <Browser>

      header={
        <>
          <GoAMicrositeHeader type="alpha" version="UAT" />

            <a href="/login">Sign in</a>
          </GoAAppHeader>
        </>
      }
      footer={<GoAAppFooter />}
      nav={
        <>
          <a href="#dashboard">Dashboard</a>

          <a href="#employees">Employees</a>
          <a href="#claims">Claims</a>
        </>
      }
      sideMenu={
        <>
          <h2>Side Menu</h2>
          <p>Add in your content here</p>
        </>

      <p>
        Lorem ipsum dolor sit amet, qui minim labore adipisicing minim sint
        cillum sint consectetur cupidatat.
